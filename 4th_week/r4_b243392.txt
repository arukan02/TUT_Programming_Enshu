  Node *x;
  if ((x = (Node *)malloc(sizeof(Node))) == NULL) {
    puts("No resource remained.");
    exit(1);
  }
  //set a default value for key
  x->key = 0;
  x->next = x;
  return x;
}

void insertNextTo(Node *x, int in_key) {
  Node *ins;
  ins = initialize();
  /* */
  ins->key = in_key;
  ins->next = x->next;
  /* */
  x->next = ins;
}

/* x should be sentinel_node */
Node *searchNodeByKey(Node *x, int key) {
  Node *search = x->next; // the target for deletion
  // stop looping if the search is <x> or the target
  while(!(search == x || search->key == key)) {
    search = search->next;
  }
  return search;
}

/* x should be sentinel_node */
void deleteKey(Node *x, int del_key) {
  Node *current = x->next;
  Node *temp = x;
  while(current != x && current->key != del_key){
    temp = current;
    current = current->next;
  }
  if (current == x) {
    printf("%d is not found.\n", del_key);
  } else {
    temp->next = current->next;
    free(current);
  }
}

/* x should be sentinel_node */
void print_list(Node *x) {
  Node *print = x->next;
  if (x->next == x) {
    puts("No data found.");
  } else {
    while(print != x) {
      printf("%d ", print->key);
      print = print->next;
    }
    putchar('\n');
  }
}

1.4. 感想 // 150字程度
4.1の課題は元の関数を変えずにやりたいので、新たなtempノード変数を使いった。
4.2の課題はsearchNodeByKeyの関数でkeyを見つけないときに最初はsentinel_nodeにしたせいで、プログラムがよく動かない。keyを見つけないときNULLにして、keyの前のポインターがsentinel_nodeの時でも、プログラムが普通に動く。
